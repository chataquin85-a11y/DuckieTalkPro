    <script>
        function switchTab(tabName) {
            document.getElementById('chat-view').style.display = 'none';
            document.getElementById('calls-view').style.display = 'none';
            document.getElementById('plan-view').style.display = 'none';
            document.getElementById('settings-view').style.display = 'none';

            document.getElementById(tabName + '-view').style.display = 'block';
            
            const titles = {
                'chat': 'Centro de Chat IA',
                'calls': 'Teclado de Llamadas',
                'plan': 'Tu Plan Móvil',
                'settings': 'Ajustes del Sistema'
            };
            document.getElementById('header-title').textContent = titles[tabName];
        }

        function sendMessage() {
            const inputField = document.getElementById('userInput');
            const chatContainer = document.getElementById('chat-container');
            const apiKeyInput = document.getElementById('apiKeyInput');
            
            const text = inputField.value.trim();
            if (!text) return;

            const userMsg = document.createElement('div');
            userMsg.className = 'message user-message';
            userMsg.textContent = text;
            chatContainer.appendChild(userMsg);
            
            inputField.value = '';
            chatContainer.scrollTop = chatContainer.scrollHeight;

            setTimeout(() => {
                const aiMsg = document.createElement('div');
                aiMsg.className = 'message ai-message';
                
                if(!apiKeyInput.value.trim()) {
                    aiMsg.textContent = "Por favor ingresa tu llave API (AIza...) en el campo correspondiente para conectar con los servidores de Duckie.";
                } else {
                    aiMsg.textContent = "Conectado correctamente. Recibido: " + text;
                }
                
                chatContainer.appendChild(aiMsg);
                chatContainer.scrollTop = chatContainer.scrollHeight;
            }, 500);
        }
    </script>
